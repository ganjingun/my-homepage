        function resetScores() { GODS.forEach(g => scores[g.id] = 0); }

        function startQuiz() {

            resetScores(); currentQ = 0;

            lockScreen.classList.add('hidden'); resultScreen.classList.add('hidden');

            quizScreen.classList.remove('hidden'); renderQuestion();

        }

        function renderQuestion() {

            if (currentQ < QUESTIONS.length) {

                const q = QUESTIONS[currentQ];

                questionDisplay.textContent = q.text;

                questionCounter.textContent = `第${currentQ+1}问 / 共${QUESTIONS.length}问`;

                optionsDiv.innerHTML = q.options.map(o => `<button class="option-btn" data-god="${o.god}">${o.text}</button>`).join('');

                document.querySelectorAll('.option-btn').forEach(b => b.addEventListener('click', onSelect));

            } else finishQuiz();

        }

        function onSelect(e) {

            const g = e.currentTarget.dataset.god;

            if (scores[g] !== undefined) scores[g]++;

            currentQ++;

            renderQuestion();

        }

        function renderMatchList(scoresObj) {

            const pairs = GODS.map(g => ({ name: g.name, score: scoresObj[g.id] || 0 }));

            pairs.sort((a,b) => b.score - a.score);

            const total = Object.values(scoresObj).reduce((s,v)=>s+v, 0) || 1;

            let html = '<div style="font-weight:500; margin-bottom:0.8rem; color:#4d4134;">神格契合度排行</div>';

            pairs.slice(0,3).forEach(p => {

                const pct = Math.round(p.score/total*100);

                html += `<div class="match-item"><span>${p.name}</span><span>${pct}%</span></div>`;

            });

            matchListContainer.innerHTML = html;

        }


        function finishQuiz() {

            let best = 'zeus', max = -1;

            for (let g of GODS) if (scores[g.id] > max) { max = scores[g.id]; best = g.id; }

            const god = GODS.find(g => g.id === best) || GODS[0];

            godNameDisp.textContent = god.name;

            godTitleDisp.textContent = god.title;

            godDescDisp.textContent = god.desc;

            const traitScores = TRAITS.map(t => {

                let raw = 0;

                t.gods.forEach(id => raw += scores[id] || 0);

                return { ...t, raw };

            });

            const totalRaw = traitScores.reduce((a,b) => a + b.raw, 0) || 1;

            let traitsHtml = '';

            traitScores.forEach(t => {

                const pct = Math.round(t.raw / totalRaw * 100);

                traitsHtml += `<div class="trait-item"><span class="trait-label">${t.name}</span><div class="trait-bar-bg"><div class="trait-bar-fill" style="width:${pct}%"></div></div><span class="trait-percent">${pct}%</span></div>`;

            });

            traitsContainer.innerHTML = traitsHtml;

            extendedAnalysisDiv.innerHTML = `<p style="margin-bottom:0.5rem;">天命评语</p><p>${generateExtendedText(traitScores)}</p>`;

            renderMatchList(scores);

            quizScreen.classList.add('hidden');

            resultScreen.classList.remove('hidden');

        }


        function resetLock() {

            pwInput.value = ''; lockError.textContent = ''; unlocked = false;

            quizScreen.classList.add('hidden'); resultScreen.classList.add('hidden');

            lockScreen.classList.remove('hidden');

        }


        unlockBtn.addEventListener('click', () => {

            if (pwInput.value.trim() === FIXED_PASSWORD) { unlocked = true; lockError.textContent = ''; startQuiz(); }

            else { lockError.textContent = '口令错误，无法进入神域'; pwInput.value = ''; }

        });

        pwInput.addEventListener('keypress', e => { if (e.key === 'Enter') unlockBtn.click(); });

        restartQuizBtn.addEventListener('click', () => { if (unlocked) startQuiz(); else resetLock(); });

        restartFromResultBtn.addEventListener('click', () => { if (unlocked) startQuiz(); else resetLock(); });

        backToLock.addEventListener('click', resetLock);

        document.addEventListener('contextmenu', e => e.preventDefault());

        resetLock();

    })();
